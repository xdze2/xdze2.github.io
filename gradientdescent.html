def onestep( x, y ):
    
    # Gradient
    dx = terrain.ev(x, y, dx=1, dy=0)
    dy = terrain.ev(x, y, dx=0, dy=1)
    
    # Linesearch
    fscan = lambda  t :  terrain.ev(x -dx*t, y -dy*t )
    res = fmin( fscan, 0.01, disp=0 )
    
    tmin =  res[0]
    return  x -dx*tmin, y -dy*tmin

    return (1-X)**2 + 100*( Y - X**2 )**2
