multi = []
A = Z[0]

same = False
for B in Z[1:]:
    if A[0]==B[0] and same==False:
        same = True
        famille = [A[1], B[1]] 
    elif A[0]==B[0] and same==True:
        famille.append( B[1] )        
    elif A[0]!=B[0] and same == True:
        multi.append(famille)
        same = False

    A = B
